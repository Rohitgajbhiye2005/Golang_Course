package basics

import (
	"fmt"
	"net/http"
)

func main() {
	fmt.Println("Hello go standard library")
	resp, err := http.Get("https://jsonplaceholder.typicode.com/posts/1")
	if err != nil {
		fmt.Println("Error found")
	}
	defer resp.Body.Close()

	fmt.Println("The response code is ", resp.Status)
}
