package main

import "fmt"

func main() {
	// panic(interface{})
	// with the valid input
	// process(1)

	// with the panic
	process(-1)

	// defer still excuted when the panic happened

	// the order is code defer then panic

}

func process(input int) {
	defer fmt.Println("deferred 1")
	defer fmt.Println("deferred 2")
	if input < 0 {
		defer fmt.Println("deferred before panic")
		panic("The input should be non negative")
		// defer fmt.Println("deffered after panic") this is unreachedble code soits not executed
	}
	fmt.Println("Processing", input)
}
